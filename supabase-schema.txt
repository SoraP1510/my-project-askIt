-- ---------- categories (เทียบเท่า subreddit) ----------
create table if not exists public.categories (
  id          bigint generated by default as identity primary key,
  topic       text not null unique,
  created_at  timestamptz not null default now()
);

-- ---------- post ----------
create table if not exists public.post (
  id            bigint generated by default as identity primary key,
  title         text not null,
  body          text,
  image         text,
  categories_id bigint references public.categories(id) on delete set null,
  username      text not null,
  created_at    timestamptz not null default now()
);
create index if not exists post_categories_id_idx on public.post(categories_id);
create index if not exists post_created_at_idx   on public.post(created_at desc);

-- ---------- comment ----------
create table if not exists public.comment (
  id          bigint generated by default as identity primary key,
  post_id     bigint not null references public.post(id) on delete cascade,
  username    text not null,
  text        text not null,
  created_at  timestamptz not null default now()
);
create index if not exists comment_post_id_idx on public.comment(post_id);

-- ---------- vote ----------
create table if not exists public.vote (
  id         bigint generated by default as identity primary key,
  post_id    bigint not null references public.post(id) on delete cascade,
  username   text not null,
  upvote     boolean not null,
  created_at timestamptz not null default now(),
  unique (post_id, username)
);
create index if not exists vote_post_id_idx on public.vote(post_id);

-- =====================================================
-- RLS policies
-- The app calls Supabase with NEXT_PUBLIC_SUPABASE_ANON_KEY
-- (no Supabase Auth — auth is via NextAuth/Google), so anon
-- needs full read/write access. Tighten this for production.
-- =====================================================

alter table public.categories enable row level security;
alter table public.post      enable row level security;
alter table public.comment   enable row level security;
alter table public.vote      enable row level security;

create policy "anon all categories" on public.categories
  for all to anon using (true) with check (true);
create policy "anon all post" on public.post
  for all to anon using (true) with check (true);
create policy "anon all comment" on public.comment
  for all to anon using (true) with check (true);
create policy "anon all vote" on public.vote
  for all to anon using (true) with check (true);
